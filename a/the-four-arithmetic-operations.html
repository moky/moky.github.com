最近在写 SnowCat 的 iOS 版，由于模版里大量用到各种参数， 之前写精灵森林的时候，那些参数通常都是固定值，最多支持几个环境变量替换即可， 这次我想支持更多。。。 另一方面，精灵森林由于是基于 Cocos2d 图形库的，在适配方面没有太大问题， 而 SnowCat-iOS 是基于 UIKit 的，所以就遇到了计算的需求。 反正是躲不掉了，所以还是支持一下表达式吧。。。。 我嫌网上那些实现代码太过复杂，而我需要的是简单高效的四则运算解析， 所以就花了点时间用 C 语言实现了一个。 因此就有了下面这个副产品，这里 share 一下给大家。 BTW: 前些天看“最强大脑”时忽然萌生一个念头， 也即兴写了一枚个人觉得还蛮好玩的小玩意儿：超大自然数幂运算， 一并分享给大家 ^_^ -------- 下面是“四则运算解析器”的源代码 -------- (arith.h/c) // // arith.h // #include <stdlib.h> #include <string.h> double calculate(const char * str); // // arith.c // #include "arith.h" #ifndef MAXFLOAT #define MAXFLOAT 0x1.fffffep+127f #endif static double eval(const char * buffer, int offset, int * bounds); static double node(const char * buffer, int offset, int * bounds) { int pos; if (buffer[offset] == '(') { return eval(buffer, offset + 1, bounds); // eval sub expression } if (buffer[offset] == '-') { return -node(buffer, offset + 1, bounds); // format: '-(x)' } for (pos = offset + 1; pos < *bounds; ++pos) { if (buffer[pos] == '+' || buffer[pos] == '-' || buffer[pos] == '*' || buffer[pos] == '/' || buffer[pos] == ')') { *bounds = pos; break; } } return atof(buffer + offset); } static double eval(const char * buffer, int offset, int * bounds) { int pos = *bounds; float left = node(buffer, offset, &pos), right; // primary operations do { if (pos >= *bounds) { return left; // finished } if (buffer[pos] == ')') { *bounds = pos + 1; return left; // end sub expression } offset = pos + 1; if (buffer[pos] == '*') { pos = *bounds; left *= node(buffer, offset, &pos); continue; } if (buffer[pos] == '/') { pos = *bounds; right = node(buffer, offset, &pos); left = (right == 0 ? MAXFLOAT : left / right); continue; } break; } while (1); // secondary operations if (buffer[pos] == '+') { return left + eval(buffer, offset, bounds); } else if (buffer[pos] == '-') { return left + eval(buffer, offset - 1, bounds); } // error return MAXFLOAT; } double calculate(const char * str) { size_t len = strlen(str); char * buffer = (char *)malloc(len + 1); int i, j; double result; // remove white spaces bzero(buffer, len + 1); for (i = 0, j = 0; i < len; ++i) { if (str[i] != ' ' && str[i] != '\t') { buffer[j] = str[i]; ++j; } } // calculate result = eval(buffer, 0, &j); free(buffer); return result; } 测试程序： (main.c) #include <stdio.h> #include "arith.h" int main(int argc, char * argv[]) { double result; if (argc != 2) { printf("\n" " Usage:\n" "\n" " %s \"1 + (2 - 3) * 4 / 5\"\n" "\n", argv[0]); return 1; } result = calculate(argv[1]); printf("\n" " Result: %lf\n" "\n", result); return 0; }
